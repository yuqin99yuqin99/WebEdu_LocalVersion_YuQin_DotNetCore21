namespace SemanticKernel_AI_LLM_Try.Controllers

{

    public class AskGPTTranslationController : Controller

    {

        [HttpGet(Name = "AskGPTTranslation")]

        public async Task<string> AskGPTTranslation(String inputString)

        {

//inputString在此传入参数是System.String字符串对象类型的对象实例"教育数据框架驱动的教师机器人辅助的教育决策支持系统"。

IKernel iKernel = Kernel.Builder.Build();//建造一个IKernel对象类型的iKernel对象实例

            ISKFunction llm = iKernel.CreateSemanticFunction("请扮演教育决策支持系统专家，告诉我"+inputString+"的涵义。"); //创建一个SemanticFunction实现模型的调用。SemanticFunction函数的输入是AIGC宿主的中英双语等等Prompt语言（目前中英双语等等Prompt语言最好是遵守RICIO语法框架）。在此的中英双语等等Prompt语言调用AIGC宿主返回回答Answer。当然也可进一步调用其他各种功能。

            SKContext? responseFromLLM = await llm.InvokeAsync(inputString);

            return responseFromLLM.Result;

        }

    }

}
